\begin{align*}
    \text{Spur}(BC) &= \sum_{j = 1}^n[BC]_{jj} = \\
                    &= \sum_{j = 1}^n\sum_{l = 1}^n[B]_{j, l} \cdot [C]_{l, j} = \sum_{l = 1}^n\sum_{j = 1}^n[B]_{j, l} \cdot [C]_{l, j} = \sum_{j = 1}^n\sum_{l = 1}^n[B]_{l, j} \cdot [C]_{j, l} = \\
                    &= \sum_{j = 1}^n[CB]_{jj} = \text{Spur}(CB) \ \ \ \square
\end{align*}
