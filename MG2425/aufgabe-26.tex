            \end{array}

        \begin{equation*}
            \begin{array}{rcl}
                f(x_1) &=& f(x_2) \quad (x \neq 2, x_1 \neq x_2) \\[5pt]
                \dfrac{3}{(2 - x_1)^2} &=& \dfrac{3}{(2 - x_2)^2} \\[10pt]
